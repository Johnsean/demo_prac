#### js:

``` javascript
// 三种状态
function current(x){   //显示
    $(x).addClass('current').removeClass('enter')
}
function leave(x){    //离开
    $(x).addClass('leave').removeClass('current')
return $(x);
}
function enter(x){  // 预备显示
    x.removeClass('leave').addClass('enter')
}
```
